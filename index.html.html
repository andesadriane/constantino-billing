async function runExport(items, filename, isAll){

  const el=buildExportEl(items, isAll);
  stg.innerHTML='';
  stg.appendChild(el);
  await new Promise(r=>setTimeout(r,150));
  const c=getColors();
  try{
    const canvas=await html2canvas(el,{scale:2.5,backgroundColor:c.bg,useCORS:true,logging:false,allowTaint:false});
    const a=document.createElement('a');
    a.download=filename+'.png';
    a.href=canvas.toDataURL('image/png');
    a.click();
  }catch(err){console.error(err);alert('Export failed. Please try again.')}
  finally{stg.innerHTML=''}
}
